چاپونه کجا

ایمان فلاح
زمان آهنگ: 3 دقیقه و 31 ثانیه
تاریخ: اردیبهشت 31، 1393

متن موسیقی:

دِ تا آهو بیمی دیمی بیابون ای خدا ... مه یار مه جا نَوه جِدا
"2تا آهو بودیم تو بیابون ای خدا ... یارم ازم جدا نشه"

اَتی لار دَرِمی اَتی مازِرون ای خدا ... مه یار مه جا نَوه جِدا
"یک کم آزادیم و یک کم مازندران هستیم ای خدا ... یارم ازم جدا نشه"

خدا خدا خامه شِه جان ِ یارِمٌه ته جا ... مه یار مه جا نَوه جِدا
"از خدا میخوام جانمو برات بیارم ... یارم ازم جدا نشه"

اَنٌه خومه مِن تِسِه تِرِه بَرِسه مه صِدا ... مه یار مه جا نَوه جِدا
"برات میخونم که صدام به تو برسه ... یارم ازم جدا نشه"


چَپونه کیجا شونه صَرا خانه هاده گِسفن چرا ... روزا هَسه صَرا وِنه جا
"دختر چوپون میره صحرا میخواد گوسفند رو چرا بده ... روزا جاش صحرائه"

خامه بورِم صَرا بنفشه بَچینِم ... های های
"میخوام برم صحرا گل بنفشه بچینم ... های های"

چَپونه کیجاره بوینم ... های های
"دختر چوپون رو ببینم ... های های"

چَپونه کیجا ته ناز ِ بِلاره ... های های
" دختر چوپون قربون نازت (یکی از اصطلاحات مازندرانی برای ناز دادن) ... های های"

ته دیمه خال ِ بلاره ... های های
" قربون خال صورتت ... های های"


هِراز ِ او بِمو پِله بَورده ... مِنو مه دِلبَره دِل ِ بَورده
" آب رود هراز پل رو برد (خراب کرد) ... دل من و دلبرم رو برد"

رفیقون جَم بَوین پِله بِسازین ... مِنو مه دِلبَر ِ دِل ِ بِسازین
" دوستان جمع بشین پل رو بسازین ... دل من و دلبرم رو بسازین"


هِوار هِوار هاکِنِم دل ِ بیقرار هاکِنِم
"دا و بیداد بکنم دلم رو بیقرار کنم"

شه یار ِ بال ِ دِماسِم اَمشو فِرار هاکِنِم
"دست یارم رو بگیرم و امشب فرار کنم"

هِوار هِوار هاکِنِم دل ِ بیقرار هاکِنِم
"داد و بیداد بکنم و دلم رو بیقرار کنم"

شه یار ِ بال ِ دِماسِم اَمشو فِرار هاکِنِم
"دست یارم رو بگیرم و امشب فرار کنم"


من اینتا اونتایه دا
" قربون این و اونم"

شِمال ِ هوای ِ دا
" قربون هوای شمال"

" قربون همزه ام"

" قربون تمام شهرها "

" قربون محمود آبادم "

" قربون فریدون کنار "

" قربون نور و رویانم "

بهشر و گرگان ِ دا
" قربون بهشهر و گرگان"

آمل و بابل ِ دا
" قربون آمل و بابل"

قائمشهر ِ پل ِ دا
" قربون پل قائمشهر"

چالوس و نوشهر ِ دا
" قربون چالوس و نوشهر"

متل قو رامسر ِ دا
" قربون متل قو و رامسر"


نکاء و ساری ِ دا
" قربون نکا و ساری"

اون شهسواری ِ دا
"قربون اون شهسوار"

میرم ِ میرم ِ
" میمیرم براشون"
اَلِنگو طِلا گیرمه چِشٌه قِربون
"برات النگو طلا میگیرم"

ته وِسه الله گیرمه تِره وَرمه
" برات الله میگیرم و تو رو میبرم"


انار ِ دِبره گیرمه اَقوز ِ چِنگه گیرمه
" یه عالمه برات انار و گردو میگیرم"

شاتوت ِ شیره گیرمه تِره قِربون
" آب شاتوت رو برات میگیرم قربونت میشم"

ته وِسه الله گیرمه تِره وَرمه
" برات الله میگیرم و تو رو میبرم"

اگه پِل ندارمه مار ِ جا گیرمه
" اگه پول ندارم از مادرم میگیرم"


اگه مار نِدائه پِر ِ جا گیرمه
" اگه مادرم نداد از پدرم میگیرم"

" از با رضایت نداد به زور میگیرم"

" میگیرم ، میگیرم"
